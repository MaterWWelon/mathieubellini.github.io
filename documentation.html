4) Mettre à jour les projets (data/projects.json)

Chaque entrée JSON génère une carte. Les images référencées doivent exister dans assets/img/.

[
  {
    "anchor": "p-cost-alerts",
    "title": "Project Cost Alerts — Star Schema",
    "year": "2025",
    "role": "BI Analyst",
    "stack": "Power BI, DAX",
    "img": "p07.svg",
    "summary": "Alerte écarts coûts (≥15%), refactor en schéma en étoile"
  }
]
  • anchor : identifiant pour les liens internes (projects.html#anchor)
  • img : nom du fichier dans assets/img/ (ex. p07.svg)
  • Ajouter un projet : dupliquer un bloc, éditer les champs, commit/push

5) Compétences (barres de niveau)

HTML (exemple) :

<div class="skill" data-level="85">
  <div class="label">Power BI (DAX/Model)</div>
  <div class="bar"><div class="fill"></div></div>
  <div class="value">85%</div>
</div>

CSS minimal :

.skill{display:grid;grid-template-columns:260px 1fr 64px;align-items:center;gap:1rem}
.skill .bar{height:14px;background:rgba(255,255,255,.08);border-radius:999px;overflow:hidden}
.skill .bar .fill{width:0;height:100%;
  background:linear-gradient(90deg,#5aa0ff,#7bd6ff);
  transition:width .8s ease}

JS (appliquer le pourcentage) :

<script>
(function(){
  function setupSkills(){
    document.querySelectorAll('.skill').forEach(el=>{
      const lvl = Math.max(0, Math.min(100, parseInt(el.getAttribute('data-level'),10)||0));
      const fill = el.querySelector('.bar .fill');
      const val = el.querySelector('.value');
      if (fill) fill.style.width = lvl + '%';
      if (val)  val.textContent = lvl + '%';
    });
  }
  if (document.readyState === 'loading'){
    document.addEventListener('DOMContentLoaded', setupSkills);
  } else { setupSkills(); }
})();
</script>
